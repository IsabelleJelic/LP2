namespace PVolume

    public partial class Form1 : Form
    {
        Double raio;
        Double altura;
        public static string nome;

        public Form1()
        {
            InitializeComponent();
        }

        private void label1_Click(object sender, EventArgs e)
        {

        }

        private void textBox1_Validated(object sender, EventArgs e)
        {
            if ((textBox1.Text == "") && (textBox2.Text == ""))
                this.Close();
            else
            {

                if (!Double.TryParse(textBox1.Text, out raio))
                {
                    MessageBox.Show("Valor de raio é inválido!");
                    textBox1.Text = "";
                    textBox1.Focus();
                }
                else if (raio <= 0)
                {
                    MessageBox.Show("Raio deve ser maior que zero!");
                    textBox1.Text = "";
                    textBox1.Focus();
                }
            }
        }

        private void button2_Click(object sender, EventArgs e)
        {
            textBox2.Clear();
            textBox1.Text = "";
            textBox3.Text = String.Empty;

            textBox1.Focus();
        }

        private void button3_Click(object sender, EventArgs e)
        {
            Close();
        }

        private void button1_Click(object sender, EventArgs e)
        {
            if (Double.TryParse(textBox2.Text, out altura) && Double.TryParse(textBox1.Text, out raio))
            {
                double volume;
                volume = Math.PI * Math.Pow(raio, 2) * altura;
                textBox3.Text = volume.ToString("N2");
            }

            else
            {
                MessageBox.Show("Valores Inválidos");
                textBox1.Focus();
            }
        }
